interface

raphaël

draws vectors in SVG with JavaScript create an SVG container: 
var paper = Raphael(htmlContainer, 300, 300); // elem, width, height

var circle = paper.circle(150, 150, 100); // x, y, radius

circle.attr({
  'fill': 'orange',
  'stroke': 'blue',
  'stroke-width': 10
});

paper.path('M50,150 L150,250 L250,50') // Move to, line to...
  .attr({
    stroke: 'green',
    'stroke-width': 10
  });

paper.path('M50,150 L150,250 L250,50 z') // 'z' = close the path
  .attr('fill', 'blue');
